function resizeGame() { width = window.innerWidth; height = window.innerHeight; canvas.width = width; canvas.height = height; faceSize = Math.min(width, height) / 10; // Base face size if (isMobileDevice()) { faceSize *= 2; // Double the face size on mobile } faceSize = Math.max(Math.min(faceSize, 90), 30); // Adjusted max size to 90 to accommodate increased faceSize heartCursor.style.width = faceSize * 2 + 'px'; heartCursor.style.height = faceSize * 2 + 'px'; heartCursor.style.marginLeft = -faceSize + 'px'; heartCursor.style.marginTop = -faceSize + 'px'; // Adjust cursorCollisionRadius based on device if (isMobileDevice()) { cursorCollisionRadius = faceSize * 3; // Match big heart size (600% font-size implies ~6x scaling) } else { cursorCollisionRadius = faceSize; } // Update custom cursor size customCursor.style.width = faceSize + 'px'; customCursor.style.height = faceSize + 'px'; customCursor.style.fontSize = faceSize + 'px'; }
